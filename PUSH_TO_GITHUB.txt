         🚀 ИНСТРУКЦИЯ ПО ПУБЛИКАЦИИ НА GITHUB
═══════════════════════════════════════════════════════════════

📋 ШАГ 1: Создайте репозиторий на GitHub
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Откройте: https://github.com/new

2. Заполните:
   Repository name: advanced-user-api
   
   Description:
   Production-ready REST API with JWT authentication, Gin framework, GORM ORM, Docker, and CI/CD
   
   Visibility: Public ✅ (для портфолио)
   
   ❌ НЕ добавляйте:
   - README
   - .gitignore  
   - License
   (у вас уже всё есть)

3. Нажмите "Create repository"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🔗 ШАГ 2: Добавьте remote и запуште
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

cd advanced-user-api

git remote add origin git@github.com:AlexRodving/advanced-user-api.git
git push -u origin main

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🌐 ШАГ 3: Настройте GitHub Pages
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. На GitHub → Settings → Pages

2. Source:
   Branch: main
   Folder: /docs
   
3. Save

4. Через 2-3 минуты сайт будет доступен:
   https://alexrodving.github.io/advanced-user-api/

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🏷️ ШАГ 4: Добавьте Topics (теги)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

На странице репозитория нажмите "⚙️" рядом с About

Добавьте topics:
  golang
  gin
  gorm
  jwt
  docker
  rest-api
  postgresql
  clean-architecture
  ci-cd
